# 友元

**关键字：** `friend`

**目的：** 让一个函数或类能访问另一个类中的私有成员

**实现：**

- 全局函数做友元

  ```c++
  class Person
  {
      // 声明友元函数（表明Test函数是Person类的友元，可以访问类内部的私有成员）
      friend void Test(Person* p);
  private:
      int a;
  public:
      int b;
  };
  
  // 全局函数
  void Test(Person* p)
  {
      cout << p->a << endl; // 访问私有成员
      cout << p->b << endl; // 访问公开成员
  }
  ```

- 类做友元

  ```c++
  // 先声明有Phone这个类，后面再实现
  class Phone; 
  
  class Person
  {
  public:
      Phone* phone;
  	Person();
      Intro();
      ~Person()
      {
          if (phone != NULL)
          {
              delete phone;
          }
      }
  };
  
  class Phone
  {
      // 声明Person类作为Phone类的友元
      friend class Person;
      string name;
  public:
      Phone()
      {
          name = "手机"
      }
  };
  
  // 类外实现成员函数
  Person::Person()
  {
      phone = new Phone;
  }
  void Person::Intro()
  {
      cout << phone->name << endl; // 访问另一个类的私有成员
  }
  ```

- 成员函数做友元

  ```c++
  class Phone; 
  
  class Person
  {
  public:
      Phone* phone;
  	Person();
      Intro1();
      Intro2();
      ~Person()
      {
          if (phone != NULL)
          {
              delete phone;
          }
      }
  };
  
  class Phone
  {
      // 声明Person类的成员函数作为Phone类的友元
      friend void Person::Intro1();
      string name;
  public:
      Phone()
      {
          name = "手机"
      }
  };
  
  Person::Person()
  {
      phone = new Phone;
  }
  void Person::Intro1()
  {
      cout << phone->name << endl; // 可以访问
  }
  void Person::Intro2()
  {
      cout << phone->name << endl; // 不可以访问
  }
  ```
